Router

Configuration (pages/router.js)

export default {
  routes: [
    { path: '/', tab: 'overview' },
    { path: '/docs/:section', handler: 'DocsPage' }
  ],
  guards: [
    (to, from) => isAuthenticated() || router.navigate('/login')
  ],
  middleware: [
    (path) => console.log('Navigating to:', path)
  ]
}

Features

  • Hash-based routing (default) or HTML5 history
  • Path parameters: /users/:id
  • Query params: /users?page=1&sort=name
  • Navigation guards
  • Middleware support
  • XSS sanitization on all route params

Navigation API

// Navigate
router.navigate('/users/123')

// Query params
router.setQueryParams({ page: 2, sort: 'date' })
router.getQueryParams() // { page: '2', sort: 'date' }

// Subscribe to changes
router.subscribe((route) => {
  console.log('Now at:', route.path)
})

// Get current route
const current = router.getCurrentRoute()

Route Parameters

// Define route with params
{ path: '/posts/:id/comments/:commentId' }

// Navigate with params
router.navigate('/posts/42/comments/99')

// Access params in handler
handler(params) {
  console.log(params.id)        // '42'
  console.log(params.commentId) // '99'
}
