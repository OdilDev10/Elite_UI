$schema - Mini Zod

Define Schema

const UserSchema = $schema({
  name: $string().min(2).max(50).required(),
  email: $string().email(),
  age: $number().min(0).max(150).optional(),
  role: $enum('admin', 'user', 'guest')
})

Parse Data

const result = UserSchema.parse({
  name: 'Juan',
  email: 'juan@example.com',
  role: 'user'
})

if (result.error) {
  console.log('Errors:', result.errors)
} else {
  console.log('Valid!', result.data)
}

Available Types

$string() $number() $boolean() $array(schema) $object({}) $enum(...)

Method Chaining

$string()
  .min(2).max(50)      // Length validation
  .email()             // Format validation
  .required()          // Mandatory
  .optional()          // Can be undefined/null

$number()
  .min(0).max(100)     // Range
  .required()

$enum('a', 'b', 'c')  // Enum values
