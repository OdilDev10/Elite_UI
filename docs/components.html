Components

Reactive components with lifecycle hooks and state management

SimpleComponent

class MyComponent extends SimpleComponent {
    constructor(selector, props = {}) {
        super(selector)
        this._props = {
            title: 'Default',
            ...props
        }
    }

    onMount() {
        // Called after render
    }

    render() {
        this.el.innerHTML = `<h1>${this._props.title}</h1>`
    }
}

new MyComponent('#app', { title: 'Hello' }).mount()
